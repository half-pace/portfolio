Question 1: Design a lexical analyzer that ignores spaces, tabs, newlines, and comments

Input File (input.txt):
// This is a comment
int main() {
    int a = 10; // another comment
    /* Multi-line
       comment */
    return 0;
} 
#include <stdio.h>
#include <ctype.h>
#include <string.h>

#define MAX_IDENT_LEN 100

int is_comment = 0;

void lexical_analyzer(FILE *fp) {
    char ch, next;
    char buffer[MAX_IDENT_LEN];
    int i;
    
    while ((ch = fgetc(fp)) != EOF) {
        // Handle single-line comment
        if (ch == '/') {
            next = fgetc(fp);
            if (next == '/') {
                while ((ch = fgetc(fp)) != '\n' && ch != EOF);
                continue;
            }
            else if (next == '*') {
                is_comment = 1;
                while (is_comment) {
                    ch = fgetc(fp);
                    if (ch == '*') {
                        next = fgetc(fp);
                        if (next == '/') {
                            is_comment = 0;
                            break;
                        }
                    }
                    if (ch == EOF) break;
                }
                continue;
            }
            else {
                ungetc(next, fp);
                printf("%c", ch);
            }
        }
        
        if (is_comment) continue;
        
        // Skip whitespace
        if (ch == ' ' || ch == '\t' || ch == '\n') {
            if (ch == '\n') printf("\n");
            continue;
        }
        
        // Handle identifiers
        if (isalpha(ch) || ch == '_') {
            i = 0;
            buffer[i++] = ch;
            while (isalnum(ch = fgetc(fp)) || ch == '_') {
                if (i < MAX_IDENT_LEN - 1) buffer[i++] = ch;
            }
            buffer[i] = '\0';
            ungetc(ch, fp);
            printf("[IDENT:%s] ", buffer);
        }
        // Handle numbers
        else if (isdigit(ch)) {
            i = 0;
            buffer[i++] = ch;
            while (isdigit(ch = fgetc(fp))) {
                if (i < MAX_IDENT_LEN - 1) buffer[i++] = ch;
            }
            buffer[i] = '\0';
            ungetc(ch, fp);
            printf("[NUM:%s] ", buffer);
        }
        // Handle operators
        else if (strchr("+-*/%=<>!", ch)) {
            printf("[OP:%c] ", ch);
        }
        else {
            printf("[%c] ", ch);
        }
    }
}

int main() {
    FILE *fp = fopen("input.txt", "r");
    if (!fp) {
        printf("Error opening file\n");
        return 1;
    }
    lexical_analyzer(fp);
    fclose(fp);
    return 0;
}

Question 2: Identify whether a given line is a comment or not

#include <stdio.h>
#include <string.h>
#include <ctype.h>

int is_single_line_comment(char *line) {
    while (isspace(*line)) line++;
    return (line[0] == '/' && line[1] == '/');
}

int is_multi_line_start(char *line) {
    while (isspace(*line)) line++;
    return (line[0] == '/' && line[1] == '*');
}

int is_multi_line_end(char *line) {
    int len = strlen(line);
    for (int i = 0; i < len - 1; i++) {
        if (line[i] == '*' && line[i+1] == '/')
            return 1;
    }
    return 0;
}

int main() {
    char line[1000];
    
    printf("Enter a line: ");
    fgets(line, sizeof(line), stdin);
    
    line[strcspn(line, "\n")] = 0;
    
    if (is_single_line_comment(line))
        printf("Yes, it's a single-line comment\n");
    else if (is_multi_line_start(line))
        printf("Yes, it's the start of a multi-line comment\n");
    else if (is_multi_line_end(line))
        printf("Yes, it's the end of a multi-line comment\n");
    else
        printf("No, it's not a comment line\n");
    
    return 0;
}

#include <stdio.h>

int isValidIdentifier(char *str) {
    if (str == NULL || strlen(str) == 0)
        return 0;
    
    if (!(isalpha(str[0]) || str[0] == '_'))
        return 0;
    
    for (int i = 1; str[i]; i++) {
        if (!(isalnum(str[i]) || str[i] == '_'))
            return 0;
    }
    
    return 1;
}

int main() {

        printf("'%s' is a valid identifier\n", identifier);
    else
        printf("'%s' is NOT a valid identifier\n", identifier);
    
    return 0;
}

#include <stdio.h>
#include <string.h>

typedef struct {
    char op[5];
    char *description;
} Operator;

Operator operators[] = {
    {"+", "Addition"}, {"-", "Subtraction"},
    {"*", "Multiplication"}, {"/", "Division"},
    {"%", "Modulus"}, {"=", "Assignment"},
    {"==", "Equality"}, {"!=", "Not Equal"},
    {"<", "Less Than"}, {">", "Greater Than"},
    {"<=", "Less Than or Equal"}, {">=", "Greater Than or Equal"},
    {"&&", "Logical AND"}, {"||", "Logical OR"},
    {"!", "Logical NOT"}
};

int num_operators = sizeof(operators) / sizeof(operators[0]);

void validate_operators(char *input) {
    int i = 0;
    int len = strlen(input);
    
    printf("Input: %s\n", input);
    printf("Operators found:\n");
    
    while (i < len) {
        int found = 0;
        for (int j = 0; j < num_operators; j++) {
            int op_len = strlen(operators[j].op);
            if (i + op_len <= len && 
                strncmp(&input[i], operators[j].op, op_len) == 0) {
                printf("  '%s' - %s\n", operators[j].op, operators[j].description);
                i += op_len;
                found = 1;
                break;
            }
        }
        if (!found) i++;
    }
}

int main() {
    char input[1000];
    printf("Enter an expression: ");
    fgets(input, sizeof(input), stdin);
    input[strcspn(input, "\n")] = 0;

Question 5: Study about Lexical Analyzer Generator (LEX) and Flex

Study Notes:

LEX (Lexical Analyzer Generator):
- Developed by Mike Lesk and Eric Schmidt at Bell Labs (1975)
- Generates a lexical analyzer (scanner) from regular expression rules
- Output is C code (lex.yy.c)

FLEX (Fast Lexical Analyzer):
- GNU replacement for LEX (faster, more features)
- Backward compatible with original LEX
- Extended features: start conditions, buffer management, reentrant scanners

Installation (Ubuntu/Debian):
sudo apt-get install flex bison

Compilation:
flex program.l
gcc lex.yy.c -lfl -o program

%{

    #include <ctype.h>
    
    int is_vowel(char c) {
        c = tolower(c);
        return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
    }
    
    void convert_to_pig_latin(char *word) {
        char first = word[0];
        int len = strlen(word);

Question 38: Predictive parser for mini language

%{
    #include <stdio.h>
    #include <stdlib.h>
    
    void yyerror(char *s);
    int yylex();
%}

%token IDENTIFIER NUMBER
%token IF ELSE WHILE INT FLOAT
%token PLUS MINUS MULT DIV ASSIGN
%token LPAREN RPAREN LBRACE RBRACE SEMICOLON

%%

program     : statement_list
            ;

statement_list : statement
               | statement_list statement
               ;

statement   : assignment_stmt
            | if_stmt
            | while_stmt
            ;

assignment_stmt : IDENTIFIER ASSIGN expression SEMICOLON
                {
                    printf("Valid assignment: %s = ...\n", $1);
                }
                ;

if_stmt     : IF LPAREN expression RPAREN LBRACE statement_list RBRACE
            | IF LPAREN expression RPAREN LBRACE statement_list RBRACE ELSE LBRACE statement_list RBRACE
            ;

while_stmt  : WHILE LPAREN expression RPAREN LBRACE statement_list RBRACE
            ;

expression  : term
            | expression PLUS term
            | expression MINUS term
            ;

term        : factor
            | term MULT factor
            | term DIV factor
            ;

factor      : NUMBER
            | IDENTIFIER
            | LPAREN expression RPAREN
            ;

%%

void yyerror(char *s) {
    fprintf(stderr, "Parse error: %s\n", s);
}

int main() {
    printf("Predictive Parser for Mini Language\n");
    return yyparse();
}

Question 41: Yacc program for Calculator

Yacc File (calculator.y):

%{
    #include <stdio.h>
    #include <stdlib.h>
    
    void yyerror(char *s);
    int yylex();
    
    #define YYSTYPE double
%}

%token NUMBER
%token PLUS MINUS MULT DIV
%token LPAREN RPAREN
%token EOL

%left PLUS MINUS
%left MULT DIV
%right UMINUS

%%

input       : /* empty */
            | input line
            ;

line        : EOL
            | expression EOL { printf("= %g\n", $1); }
            ;

expression  : NUMBER          { $$ = $1; }
            | expression PLUS expression { $$ = $1 + $3; }
            | expression MINUS expression { $$ = $1 - $3; }
            | expression MULT expression { $$ = $1 * $3; }
            | expression DIV expression { 
                if ($3 == 0) {
                    yyerror("Division by zero");
                    $$ = 0;
                } else {
                    $$ = $1 / $3;
                }
            }
            | LPAREN expression RPAREN { $$ = $2; }
            | MINUS expression %prec UMINUS { $$ = -$2; }
            ;

%%

void yyerror(char *s) {
    fprintf(stderr, "Error: %s\n", s);
}

int main() {
    printf("Yacc Calculator\n");
    printf("Enter arithmetic expressions:\n");
    yyparse();
    return 0;
}
